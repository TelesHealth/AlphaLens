STEP 3 — Auth service
Create artifacts/api-server/src/services/auth.ts

Functions:
  hashPassword(password): Promise<string>
    bcryptjs saltRounds: 12

  verifyPassword(password, hash): Promise<boolean>
    bcryptjs.compare

  generateToken(userId, email, role): string
    JWT signed with process.env.JWT_SECRET
    expiresIn: "7d"
    payload: { userId, email, role }

    Returns null on any error

  encryptCredentials(credentials: object): string
    AES-256-GCM using
    process.env.CREDENTIALS_ENCRYPTION_KEY

STEP 4 — Auth middleware
Create artifacts/api-server/src/middleware/auth.ts

  requireAuth:
    Reads Bearer token from Authorization header
    OR cookie "arclion_token"
    Calls verifyToken()
    Valid: attaches req.user = { userId, email, role }
           calls next()
    Invalid: returns 401
             { error: "Authentication required" }

  requireAdmin:
    Calls requireAuth first
    role !== "admin": returns 403
    { error: "Admin access required" }

  optionalAuth:
    Same as requireAuth but always calls next()
    req.user undefined if no token

STEP 5 — Auth routes
Create artifacts/api-server/src/routes/auth.ts

POST /api/auth/register
  Body: { email, password, name }
  Zod validation:
    email: valid format
    password: min 8 chars
    name: min 2 chars
  Check email not already registered
  Hash password
  Insert user
  Generate token
  Return: { user: { id, email, name, role },
            token, message: "Account created" }

POST /api/auth/login
  Body: { email, password }
  Find user by email
  Verify password
  Wrong: 401 { error: "Invalid email or password" }
  Update lastLoginAt
  Generate token
  Set httpOnly cookie "arclion_token" 7d expiry
  Return: { user: { id, email, name, role }, token }

POST /api/auth/logout
  Clear "arclion_token" cookie
  Return: { message: "Logged out" }

GET /api/auth/me
  requireAuth middleware
  Return: { user: { id, email, name, role,
            createdAt, lastLoginAt } }

POST /api/auth/change-password
  requireAuth middleware
  Body: { currentPassword, newPassword }
  Verify current, hash and save new
  Return: { message: "Password updated" }
